值和指针

type Person struct {
  name    string
  addr    Addr
  hobbies []string
}

type Addr struct {
  country string
  detail  string
}

func NewPerson() Person{
  return Person{
    name: "hyc",
    addr: Addr{
      country: "China",
      detail:  "ShenZhen",
    },
    hobbies: []string{"eat","play","sleep"},
  }
}
入参：传值和指针的区别？
// 传值
func checkPassValue() {
  p := NewPerson()
  log.Println("------ before update ------")
  log.Println(p)
  updateByPassValue(p)
  log.Println("------ after update ------")
  log.Println(p)
}

func updateByPassValue(p Person) {
  p.name="hdr"
  p.addr.detail="GuangZhou"
  p.hobbies[1] = "study"
  log.Println("------ in update ------")
  log.Println(p)
}
// ------ before update ------
// {hyc {China ShenZhen} [eat play sleep]}
// ------ in update ------
// {hdr {China GuangZhou} [eat study sleep]}
// ------ after update ------
// {hyc {China ShenZhen} [eat study sleep]}
// 为什么hobbies的修改却能影响原来的值，因为你hobbies是个切片，底层存的是数组的指针

// 传指针
func checkPassPtr(){
  p := NewPerson()
  log.Println("------ before update ------")
  log.Println(p)
  updateByPassPtr(&p)
  log.Println("------ after update ------")
  log.Println(p)
}

func updateByPassPtr(p *Person) {
  p.name="hdr"
  p.addr.detail="GuangZhou"
  p.hobbies[1] = "study"
  log.Println("------ in update ------")
  log.Println(p)
}
// ------ before update ------
// {hyc {China ShenZhen} [eat play sleep]}
// ------ in update ------
// &{hdr {China GuangZhou} [eat study sleep]}
// ------ after update ------
// {hdr {China GuangZhou} [eat study sleep]}
// 这个结果很好理解，不例外
