AQS(AbstractQueuedSynchronizer)基于队列的同步化器。
什么是同步化器？为什么叫这个？ 待验证 同步化器用于保证一段代码是同步执行的。 所谓同步执行，就是发出一个功能调用时，会一直等待该功能执行完毕，才做后续事情。 在多线程环境下，因为CPU时间片的原因，一条线程可能运行到一半就被切换掉，导致功能只执行了一半。

AQS

AQS本质是一个资源管理器，只有获取到资源的线程才能进行下一步操作。
总的来说，AQS其实就是由三部分组成：资源状态标识、当前拥有资源的线程、记录等待资源的线程队列

等待队列

AQS的等待队列是CLH队列的变种。我们先来看看它的节点是长什么样的。
// 一个节点表示一条正在等待资源的线程
static final class Node {
        // 节点分为两种模式： 共享式和独占式
        static final Node SHARED = new Node();
        static final Node EXCLUSIVE = null;
        
        // 该节点的状态
        volatile int waitStatus;
        // 该节点已经被取消了，可能是超时或者被中断
        static final int CANCELLED =  1;
        // 当前节点释放资源或者取消时，需要通知下一个节点
        static final int SIGNAL    = -1;
        // 节点在等待Condition
        // 其他线程对Condition调用singnal()方法后，该节点加入到同步队列中。
        static final int CONDITION = -2;
        // 表示下一次共享式获取同步状态的时会被无条件的传播下去。
        static final int PROPAGATE = -3;

        // 当前节点的前后节点
        volatile Node prev;
        volatile Node next;
        
        // 该节点对应的线程
        volatile Thread thread;

        // 指向下一个等待condition的节点
        // condition只能在独占模式下使用，所以如果是共享模式，这里指向的是SHARED节点
        Node nextWaiter;
}

入队操作
入队无非就是将tail（使用CAS保证原子操作）指向新节点，新节点的prev指向队列中最后一节点（旧的tail节点），原队列中最后一节点的next节点指向新节点以此来建立联系。
// 通过指定的mode为当前线程创建一个入队节点,然后不断循环直至成功把节点插入到队尾
private Node addWaiter(Node mode) {
    Node node = new Node(mode);

    for (;;) {
        Node oldTail = tail;
        if (oldTail != null) {
            node.setPrevRelaxed(oldTail);
            if (compareAndSetTail(oldTail, node)) {
                oldTail.next = node;
                return node;
            }
        } else {
            initializeSyncQueue();
        }
    }
}
出队操作
因为这是个FIFO的队列，首节点是获取资源的节点，首节点的线程释放资源后，将会唤醒它的后继节点（next），而后继节点将会在获取资源成功时将自己设置为首节点。
public final boolean release(int arg) {
    if (tryRelease(arg)) {
        Node h = head;
        if (h != null && h.waitStatus != 0)
            unparkSuccessor(h);
        return true;
    }
    return false;
}
// 奇怪，这里没有看到设置头结点的行为呀？
// 当前节点释放资源后，下一个节点会自旋获取资源，获取成功后就会吧自己设为头节点
// 理念就是每个节点处理好自己的事情，获取到资源后，自己初始化，释放资源后，自己做清理

独占资源

AQS提供了两种使用资源的方式：独占和共享。
即同一时刻只能有一个线程获取同步状态，其它获取同步状态失败的线程则会加入到同步队列中进行等待。
独占式获取资源的顶级入口acquire(int arg)方法，如果线程获取到资源则直接返回, 否则把当前线程构造成独占式（node.EXCLUSIVE）模式节点并添加到同步队列尾部，直到获取到资源为止，整个过程忽略中断。
public final void acquire(int arg) {
    // tryAcquire(arg)：尝试获取资源、获取成功则直接返回。
    // addWaiter(Node.EXCLUSIVE)：当资源获取失败时，构建一个独占式节点并将其加入到同步队列的尾部。
    // acquireQueued(Node, arg)) : 获取该节点指定数量的资源，通过自旋的方式直到获取成功，返回是该节点线程的中断状态。
    // selfInterrupt()： 将中断补上（因其获取资源的整个过程是忽略中断的所以最后手动将中断补上）

    if (!tryAcquire(arg) &&
        acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
        selfInterrupt();
}

// 需要子类自己实现
protected boolean tryAcquire(int arg) {
        throw new UnsupportedOperationException();
}
// 通过指定的@{param}mode为当前线程创建一个入队节点,然后不断循环直至成功把节点插入到队尾
private Node addWaiter(Node mode) {
    Node node = new Node(mode);

    for (;;) {
        Node oldTail = tail;
        if (oldTail != null) {
            node.setPrevRelaxed(oldTail);
            if (compareAndSetTail(oldTail, node)) {
                oldTail.next = node;
                return node;
            }
        } else {
            initializeSyncQueue();
        }
    }
}
final boolean acquireQueued(final Node node, int arg) {
    boolean interrupted = false;
    try {
        for (;;) {
            // 获取当前节点的前驱节点
            final Node p = node.predecessor();
            // 如果其前驱节点为head 节点，说明此节点有资格去获取资源了。
            if (p == head && tryAcquire(arg)) {
                setHead(node);
                p.next = null; // help GC
                return interrupted;
            }
            // 获取资源失败，判断是否需要挂起当前线程
            if (shouldParkAfterFailedAcquire(p, node))
                interrupted |= parkAndCheckInterrupt();
        }
    } catch (Throwable t) {
        cancelAcquire(node);
        if (interrupted)
            selfInterrupt();
        throw t;
    }
}
private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
    int ws = pred.waitStatus;
    if (ws == Node.SIGNAL)
        return true;
    if (ws > 0) {
        do {
            node.prev = pred = pred.prev;
        } while (pred.waitStatus > 0);
        pred.next = node;
    } else {
        pred.compareAndSetWaitStatus(ws, Node.SIGNAL);
    }
    return false;
}
独占式释放共享资源的顶级入口release(int arg) 方法，彻底释放共享状态（state = 0）并唤醒其后继节点来获取共享状态。
public final boolean release(int arg) {
    if (tryRelease(arg)) {
        Node h = head;
        if (h != null && h.waitStatus != 0)
            unparkSuccessor(h);
        return true;
    }
    return false;
}
private void unparkSuccessor(Node node) {

    // 把节点等待状态设置为0
    int ws = node.waitStatus;
    if (ws < 0) node.compareAndSetWaitStatus(ws, 0);

    // 找到下一个非null且等待状态<=0的节点，让其对应线程休眠
    Node s = node.next;
    if (s == null || s.waitStatus > 0) {
        s = null;
        for (Node p = tail; p != node && p != null; p = p.prev)
            if (p.waitStatus <= 0)
                s = p;
    }
    if (s != null) LockSupport.unpark(s.thread);
}

ReentrantLock

ReentrantLock为例，锁其实就可以看做是一种资源，获取到锁的线程才能执行代码。
ReentrantLock内部就是使用了AQS对象用来负责获取和释放锁。
public class ReentrantLock implements Lock, java.io.Serializable {

    private final Sync sync;
    
    abstract static class Sync extends AbstractQueuedSynchronizer{...}
    
    static final class NonfairSync extends Sync{...}
    
    static final class FairSync extends Sync{...}
}
// 公平锁
static final class FairSync extends Sync {
  @ReservedStackAccess
  protected final boolean tryAcquire(int acquires) {
      final Thread current = Thread.currentThread();
      int c = getState();
      // 如果资源可用
      if (c == 0) {
          // hasQueuedPredecessors是否有线程先于当前线程，在等待资源
          if (!hasQueuedPredecessors() &&
              compareAndSetState(0, acquires)) {
              setExclusiveOwnerThread(current);
              return true;
          }
      }
      // 如果资源不可用，查看是否是当前线程拥有资源
      else if (current == getExclusiveOwnerThread()) {
          int nextc = c + acquires;
          if (nextc < 0) throw new Error("Maximum lock count exceeded");
          setState(nextc);
          return true;
      }
      return false;
  }
}
// 非公平锁
static final class NonfairSync extends Sync {
    private static final long serialVersionUID = 7316153563782823691L;
    protected final boolean tryAcquire(int acquires) {
        return nonfairTryAcquire(acquires);
    }
}

@ReservedStackAccess
final boolean nonfairTryAcquire(int acquires) {
    final Thread current = Thread.currentThread();
    int c = getState();
    if (c == 0) {
        // 资源能用就直接抢，不排队了
        if (compareAndSetState(0, acquires)) {
            setExclusiveOwnerThread(current);
            return true;
        }
    }
    else if (current == getExclusiveOwnerThread()) {
        int nextc = c + acquires;
        if (nextc < 0) // overflow
            throw new Error("Maximum lock count exceeded");
        setState(nextc);
        return true;
    }
    return false;
}

共享资源

共享资源与独占资源唯一的区别是在于同一时刻可以有多个线程获取到资源。
共享式获取资源的顶级入口是acquireShared(int arg)，如果当前线程未获取到同步状态，将会加入到同步队列中等待，与独占式唯一的区别是在于同一时刻可以有多个线程获取到资源。
public final void acquireShared(int arg) {
    // 尝试获取资源，如果获取失败
    if (tryAcquireShared(arg) < 0)
        doAcquireShared(arg);
}

// 需要子类实现
// 获取获取资源失败，返回负数
// 获取最后一份资源，返回0
// 获取资源后还有剩余，返回正数
protected int tryAcquireShared(int arg) {
    throw new UnsupportedOperationException();
}

private void doAcquireShared(int arg) {
    // 已共享模式添加到等待队列
    final Node node = addWaiter(Node.SHARED);
    boolean interrupted = false;
    try {
        for (;;) {
            final Node p = node.predecessor();
            if (p == head) {
                int r = tryAcquireShared(arg);
                if (r >= 0) {
                    setHeadAndPropagate(node, r);
                    p.next = null; // help GC
                    return;
                }
            }
            if (shouldParkAfterFailedAcquire(p, node))
                interrupted |= parkAndCheckInterrupt();
        }
    } catch (Throwable t) {
        cancelAcquire(node);
        throw t;
    } finally {
        if (interrupted)
            selfInterrupt();
    }
}
释放共享资源
public final boolean releaseShared(int arg) {
    if (tryReleaseShared(arg)) {
        doReleaseShared();
        return true;
    }
    return false;
}
