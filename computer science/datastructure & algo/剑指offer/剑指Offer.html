public boolean findNumberIn2DArray(int[][] matrix, int target) {
    if (matrix.length == 0) return false;
    int m = 0, n = matrix[0].length - 1;
    while (m < matrix.length && n >= 0) {
        int num = matrix[m][n];
        if (num > target) n--;
        else if (num < target) m++;
        else return true;
    }
    return false;
}

public int[] exchange(int[] nums) {
    int len = nums.length;
    int lo = 0 , hi = len - 1;
    while(true){
        while(lo<len&&nums[lo]%2==1) lo++;
        while(hi>=0&&nums[hi]%2==0) hi--;
        if(lo>=hi) break;
        swap(nums,lo,hi);
    }
    return nums;
}

private void swap(int[]nums,  int i , int j){
    int t = nums[i];
    nums[i] = nums[j];
    nums[j] = t;
}

public int[] spiralOrder(int[][] matrix) {
    if (matrix.length == 0) return new int[0];
    int m = matrix.length, n = matrix[0].length;
    int[] ans = new int[m * n];
    int U = 0, D = m - 1, L = 0, R = n - 1, x = 0;
    while (true) {
        for (int i = L; i <= R; i++) ans[x++] = matrix[U][i];
        if (++U > D) break;
        for (int i = U; i <= D; i++) ans[x++] = matrix[i][R];
        if (--R < L) break;
        for (int i = R; i >= L; i--) ans[x++] = matrix[D][i];
        if (--D < U) break;
        for (int i = D; i >= U; i--) ans[x++] = matrix[i][L];
        if (++L > R) break;
    }
    return ans;
}
