最值问题

问题：n个物品中选取，刚好装满背包，能得到的最大价值
dp[i][j] 前i个物品中选取，体积恰好为j时，最大的价值
如果第i个物品不取，则dp[i][j]=dp[i-1][j]
如果第i个物品取，则dp[i][j]=dp[i-1][j-v[i]]+w[i]
dp[i][j] = max(dp[i-1][j],dp[i-1][j-v[i]]+w[i])
public int zeroOnePackage(int[] w, int[] v, int capacity) {
  int quantity = w.length;
  int[][] dp = new int[quantity][capacity + 1];
  // dp[i][j] = 0~i的物品中选取,容量刚好为j的最大价值
  // dp[i][j] = max(dp[i-1][j],dp[i-i][j-v[i]]+w[i])

  // 初始化
  dp[0][v[0]] = w[0];

  for (int i = 1; i < quantity; ++i) {
    for (int j = capacity; j>=0; --j) {
      dp[i][j] = dp[i - 1][j];
      if (j >= v[i]) {
        dp[i][j] = Math.max(dp[i][j], dp[i - 1][j - v[i]] + w[i]);
      }
    }
  }

  return dp[quantity - 1][capacity];
}

public int zeroOnePackage(int[] w, int[] v, int capacity) {
  int quantity = w.length;
  int[][] dp = new int[quantity][capacity + 1];
  // dp[i][j] = 0~i的物品中选取,容量不超过j的最大价值
  // dp[i][j] = max(dp[i-1][j],dp[i-i][j-v[i]]+w[i])

  // 初始化 dp[0][j] = j>=v[0] ? w[0] : 0
  for(int j=capacity;j>=v[0];--j){
    dp[0][j] = w[0];
  }

  for (int i = 1; i < quantity; ++i) {
    for (int j = capacity; j>=0; --j) {
      if (j >= v[i]) {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i - 1][j - v[i]] + w[i]);
      }else{
        dp[i][j] = dp[i - 1][j];
      }
    }
  }

  return dp[quantity - 1][capacity];
}

public int zeroOnePackage(int[] w, int[] v, int capacity) {
  int quantity = w.length;
  int[] dp = new int[capacity + 1];
  for(int j=capacity;j>=v[0];--j){
    dp[j] = w[0];
  }

  for (int i = 1; i < quantity; ++i) {
    for (int j = capacity; j>=v[i]; --j) {
      dp[j] = Math.max(dp[j], dp[j - v[i]] + w[i]);
    }
  }

  return dp[quantity - 1][capacity];
}
