public int maxProduct(int[] nums) {
    int ans = Integer.MIN_VALUE, max = 1, min = 1;
    for (int num : nums) {
        int oldMax = max;
        max = Math.max(num, num > 0 ? max * num : min * num);
        min = Math.min(num, num > 0 ? min * num : oldMax * num);
        ans = Math.max(ans, max);
    }
    return ans;
}

public int longestWPI(int[] hours) {
    int ans = 0;
    int score = 0;
    int len = hours.length;
    Map<Integer, Integer> seen = new HashMap<>();
    for (int i = 0; i < len; i++) {
        score += hours[i] > 8 ? 1 : -1;
        if (score > 0) {
            ans = i + 1;
        } else {
            seen.putIfAbsent(score, i);
            if (seen.containsKey(score - 1)) {
                ans = Math.max(ans, i - seen.get(score - 1));
            }
        }
    }
    return ans;
}
