  public int findLengthOfLCIS(int[] nums) {
      int len = nums.length;
      int[] dp = new int[len];
      dp[0] = 1;
      int ans = 1;
      for(int i=1;i<len;++i){
          dp[i] = nums[i]>nums[i-1] ? dp[i-1]+1 : 1;
          ans = Math.max(ans,dp[i]);
      }
      return ans;
  }

152. 乘积最大子数组
从第一个元素开始不停的累乘，分别记录最小值和最大值。因为负数会让最小值变成最大值，最大值变成最小值，所以遇到负数时，需要先把最大值最小值进行交换。
// n[i]表示i结束乘积为负数且乘积最小子数组的乘积 p[i]表示i结尾乘积的最大子数组的乘积
// p[i]=max(num,num>0?p[i-1]*num:n[i-1]*num)
// n[i]=min(num,num>0?n[i-1]*num:p[i-1]*num)
public int maxProduct(int[] nums) {
    int ans = Integer.MIN_VALUE, max = 1, min = 1;
    for (int num : nums) {
        int oldMax = max;
        max = Math.max(num, num > 0 ? max * num : min * num);
        min = Math.min(num, num > 0 ? min * num : oldMax * num);
        ans = Math.max(ans, max);
    }
    return ans;
}

1124. Longest Well-Performing Interval
按照一天工作时间大于8小时(福报日)就加1分，否则(摸鱼日)就减1分的规则，我们能得到一个得分数组scores。scores[i]表示[0,i]福报日与摸鱼日的天数差。
当scores[i]>0，说明[0,i]福报日多于摸鱼日，则表现良好最长时间段是[0,i]，结果是i+1。
当scores[i]≤0，假设scores[i]=-1，因为我们要福报日多于摸鱼日(scores[i]>0)，也就是要少摸鱼2天，唯一办法就是少统计摸鱼的2天
public int longestWPI(int[] hours) {
    int ans = 0;
    int score = 0;
    int len = hours.length;
    Map<Integer, Integer> seen = new HashMap<>();
    for (int i = 0; i < len; i++) {
        score += hours[i] > 8 ? 1 : -1;
        if (score > 0) {
            ans = i + 1;
        } else {
            seen.putIfAbsent(score, i);
            if (seen.containsKey(score - 1)) {
                ans = Math.max(ans, i - seen.get(score - 1));
            }
        }
    }
    return ans;
}
