public List<String> letterCombinations(String digits) {

    return ans;

private void dfs(String digits, int idx) {
    if (idx == digits.length()) {
        ans.add(sb.toString());
        return;
    }

    if (digits.charAt(idx) == '1') dfs(digits, idx + 1);

        dfs(digits, idx + 1);
        sb.deleteCharAt(sb.length() - 1);
    }
}
