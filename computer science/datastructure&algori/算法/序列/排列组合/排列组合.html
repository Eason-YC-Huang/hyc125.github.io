46. 全排列
private List<List<Integer>> ans = new ArrayList<>();
private List<Integer> sub = new ArrayList<>();
private boolean[] used;

public List<List<Integer>> permute(int[] nums) {
    used = new boolean[nums.length];
    backtrace(nums, 0);
    return ans;
}

private void backtrace(int[] nums, int idx) {
    if (idx == nums.length) {
        ans.add(new ArrayList<>(sub));
        return;
    }

    for (int i = 0; i < nums.length; i++) {
        if (used[i]) continue;
        sub.add(nums[i]);
        used[i] = true;
        backtrace(nums, idx + 1);
        used[i] = false;
        sub.remove(sub.size() - 1);
    }
}
47. 全排列 II
List<List<Integer>> ans = new ArrayList<>();
List<Integer> sub = new ArrayList<>();
boolean[] used;

public List<List<Integer>> permuteUnique(int[] nums) {
    Arrays.sort(nums);
    used = new boolean[nums.length];
    backtrack(nums);
    return ans;
}

public void backtrack(int[] nums) {
    if (sub.size() == nums.length) {
        ans.add(new ArrayList<>(sub));
        return;
    }
    for (int i = 0; i < nums.length; ++i) {
        if (used[i]) continue;
        if (i > 0 && nums[i] == nums[i - 1] && !used[i - 1]) continue;

        sub.add(nums[i]);
        used[i] = true;
        backtrack(nums);
        used[i] = false;
        sub.remove(sub.size() - 1);
    }
}
526. 优美的排列
int ans = 0;
boolean[] used;

public int countArrangement(int n) {
    used = new boolean[n + 1];
    backtrace(n, 1);
    return ans;
}

private void backtrace(int n, int idx) {
    if (idx > n) {
        ans++;
        return;
    }
    for (int i = 1; i <= n; i++) {
        if (used[i]) continue;
        if (idx % i == 0 || i % idx == 0) {
            used[i] = true;
            backtrace(n, idx + 1);
            used[i] = false;
        }
    }
}

17. 电话号码的字母组合
static String[] dict = {"abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};

List<String> ans = new ArrayList<>();
StringBuilder sb = new StringBuilder();

public List<String> letterCombinations(String digits) {
    dfs(digits, 0);
    return ans;
}

private void dfs(String digits, int idx) {
    if (idx == digits.length()) {
        ans.add(sb.toString());
        return;
    }

    if (digits.charAt(idx) == '1') dfs(digits, idx + 1);
    String options = dict[digits.charAt(idx) - '2'];
    for (int i = 0; i < options.length(); i++) {
        sb.append(options.charAt(i));
        dfs(digits, idx + 1);
        sb.deleteCharAt(sb.length() - 1);
    }
}
