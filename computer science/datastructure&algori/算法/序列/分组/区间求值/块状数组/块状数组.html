public class BlockArray {

    private static class Block {
        int lo, hi;
        int sum;
        int lzy;

        public Block() {
        }

        public Block(int lo, int hi) {
            this.lo = lo;
            this.hi = hi;
        }

        @Override
        public String toString() {
            return "{" + lo + "," + hi + "," + sum + "}";
        }
    }

    int[] arr;
    Block[] blocks;
    int[] id;

    public BlockArray(int[] nums) {
        this.arr = nums;
        int len = nums.length;
        // è®¡ç®—å—çš„ä¸ªæ•°,åˆå§‹åŒ–æ¯ä¸ªå—è®°å½•å…ƒç´ çš„å¼€å§‹å’Œç»“æŸä½ç½®
        this.blocks = new Block[(int) Math.sqrt(len)];
        for (int i = 0; i < blocks.length; i++) {
            int lo = len / blocks.length * (i);
            int hi = len / blocks.length * (i + 1) - 1;
            blocks[i] = new Block(lo, hi);
        }
        // ç”±äºæ•°ç»„çš„é•¿åº¦å¹¶ä¸ä¸€å®šæ˜¯ä¸€ä¸ªå®Œå…¨å¹³æ–¹æ•°ï¼Œé‚£ä¹ˆæˆ‘ä»¬æŠŠé—æ¼ä¸‹çš„é‚£ä¸€å°å—æ”¾å…¥æœ€åä¸€ä¸ªå—ä¸­
        blocks[blocks.length - 1].hi = len - 1;

        // ä¸ºæ¯ä¸ªå…ƒç´ ç¡®å®šå…¶æ‰€å±çš„å—
        // å¦‚æœéœ€è¦ï¼Œæˆ‘ä»¬è¿˜å¯ä»¥é¢„å¤„ç† e.g. ç»Ÿè®¡æ¯ä¸ªå—çš„å¤§å°,ç»Ÿè®¡æ¯ä¸ªå—çš„åˆ
        this.id = new int[len];
        for (int i = 0; i < blocks.length; i++) {
            int sum = 0;
            for (int j = blocks[i].lo; j <= blocks[i].hi; j++) {
                id[j] = i;
                sum += nums[j];
            }
            blocks[i].sum = sum;
        }
    }
    
    /**
     * åŒºé—´ä¿®æ”¹
     * å½“ ğ‘¥ ä¸ ğ‘¦ åœ¨åŒä¸€å—å†…æ—¶ï¼Œç›´æ¥æš´åŠ›ä¿®æ”¹åŸæ•°ç»„ï¼Œç„¶åæ›´æ–°æ‰€åœ¨å—çš„ğ‘ ğ‘¢ğ‘š

    public void add(int x, int y, int k) {
        if (id[x] == id[y]) {
            for (int i = x; i <= y; i++) {
                arr[i] += k; blocks[id[x]].sum += k;
            }
        } else {
            for (int i = x; i <= blocks[id[x]].hi; i++) {
                arr[i] += k;
                blocks[id[x]].sum += k;
            }
            for (int i = y; i >= blocks[id[y]].lo; i--) {
                arr[i] += k;
                blocks[id[x]].sum += k;
            }
            for (int i = id[x] + 1; i < id[y]; i++) blocks[i].lzy += k;
        }
    }

    /**
     * åŒºé—´æŸ¥è¯¢
     * å¦‚æœå·¦å³ä¸¤è¾¹åœ¨åŒä¸€å—ï¼Œç›´æ¥æš´åŠ›è®¡ç®—åŒºé—´å’Œ

     */
    public int query(int x, int y) {
        int ans = 0;
        if (id[x] == id[y])
            for (int i = x; i <= y; i++)
                ans += arr[i] + blocks[id[i]].lzy;
        else{
            for (int i = x; i <= blocks[id[x]].hi; i++) ans += arr[i] + blocks[id[i]].lzy;
            for (int i = y; i >= blocks[id[y]].lo; i--) ans += arr[i] + blocks[id[i]].lzy;
            for (int i = id[x] + 1; i < id[y]; i++) ans += blocks[i].sum + blocks[i].lzy * (blocks[i].hi - blocks[i].lo + 1);
        }
        return ans;
    }
    
}
