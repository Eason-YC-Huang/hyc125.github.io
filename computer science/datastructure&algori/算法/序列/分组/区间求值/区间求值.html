  • 数组不变，区间查询：前缀和、树状数组、线段树、块状数组；
  • 数组单点修改，区间查询：树状数组、线段树、块状数组；
  • 数组区间修改，单点查询：差分、线段树、块状数组；
  • 数组区间修改，区间查询：线段树、块状数组；

public boolean containsNearbyAlmostDuplicate(int[] nums, int k, int t) {
    // |nums[i] - nums[j]| <= t
    // nums[j]-nums[i]>=-t && nums[j]-nums[i]<=t
    // nums[j]+t>=nums[i] && nums[j]-t<=nums[i];
    int len = nums.length;
    TreeMap<Long, Integer> tree = new TreeMap<>();

    for (int lo = 0, hi = 0; hi < len; hi++) {
        if (hi - lo > k) tree.remove((long) nums[lo++]);
        long num = nums[hi];
        Map.Entry<Long, Integer> entry = tree.floorEntry(num + t);
        if (entry != null && Math.abs(hi - entry.getValue()) <= k && num - t <= entry.getKey()) return true;
        tree.put(num, hi);
    }
    return false;
}

public boolean increasingTriplet(int[] nums) {
    // 求最长递增子序列是否大于等于3
    // 使用一条单调队列
    // 遍历nums,假设当前为x,用x替换队列中第一个大于等于它的元素
    // 其实它的本质是贪心,用更小的元素去替换第一个比它大的元素,那么就能增大后面的元素进入队列的机会
    int[] q = new int[3];
    int idx = 0;
    for (int num : nums) {
        int lo = 0, hi = idx;
        while (lo < hi) {
            int mid = (hi + lo) >> 1;
            if (q[mid] >= num) hi = mid;
            else lo = mid + 1;
        }
        q[lo] = num;
        if (lo == idx) idx++;
        if (idx == 3) return true;
    }
    return false;
}

这里有一点要注意，在Java里面，我们常常以为%就是求模运算，但其实不是的，%其实是求余运算，两者是有区别的。 对于整型数a，b来说，取模运算或者求余运算的方法都是： 1.求整数商： c = [a/b] 2.计算模或者余数： r = a - c*b. 求模运算和求余运算在第一步不同： - 求余运算在取c的值时，向0 方向舍入 - 取模运算在取c的值时，向负无穷方向舍入 当a和b正负号一致时，求模运算和求余运算所得的c的值一致，因此结果一致。 当正负号不一致时，结果不一样。 在各个环境下%运算符的含义是不同的，比如c/c++，java 为取余，而python则为取模。
问题来了，在Java下怎么实现负数取模呢？ 使用Math.floorMod函数。

public int longestWPI(int[] hours) {

    for (int i = 1; i < sum.length; i++)
        sum[i] = sum[i - 1] + (hours[i - 1] > 8 ? 1 : -1);

    int ans = 0;
    Map<Integer, Integer> seen = new HashMap<>();
    for (int i = 1; i < sum.length; i++) {
        if (sum[i]>0) ans = i;
        else {
            Integer idx = seen.get(sum[i] - 1);
            if (idx != null) ans = Math.max(ans, i - idx);
            seen.putIfAbsent(sum[i], i);
        }
    }
    return ans;
}

public long[] getDistances(int[] arr) {
    // arr中相同的元素一组
    // 对于每一组可以理解为,给你一个nums数组,要求返回abs数组,
    // abs[i] = nums[i]与其它元素的绝对值之和
    // abs[i] = nums[i]-nums[0] + nums[i]-nums[1] + ... + nums[i]-nums[i]
    //          + nums[i+1]-nums[i] + ... + nums[n-1] -nums[i]
    // abs[i] = (i+1)*nums[i] - ∑[0,i] + ∑[i+1,n-1]-(n-i-1)*nums[i]
    int len = arr.length;
    Map<Integer, List<Integer>> groups = new HashMap<>();
    for (int i = 0; i < arr.length; i++) groups.computeIfAbsent(arr[i], key -> new ArrayList<>()).add(i);
    long[] ans = new long[len];
    for (List<Integer> nums : groups.values()) absSum(nums, ans);
    return ans;
}

private void absSum(List<Integer> nums, long[] ans) {
    int n = nums.size();
    // 这里要用long,被坑惨了
    long[] sum = new long[n + 1];
    for (int i = 1; i < sum.length; i++)
        sum[i] = sum[i - 1] + nums.get(i - 1);

    for (int i = 0; i < n; i++) {
        long a = (long) (i + 1) * nums.get(i) - (sum[i + 1] - sum[0]);
        long b = (sum[n] - sum[i + 1] - (long) (n - i - 1) * nums.get(i));
        ans[nums.get(i)] = a + b;
    }
}

private int maxSumSubArr(int[] arr, int k) {
    int len = arr.length;

    int ans = Integer.MIN_VALUE;
    int[] sum = new int[len + 1];
    for (int i = 1; i <= len; i++) sum[i] = sum[i - 1] + arr[i - 1];

    for (int i = 0; i < len; i++) {
        for (int j = i; j < len; j++) {
            int cur = sum[j + 1] - sum[i];
            if (cur<=k) ans = Math.max(ans, cur);
        }
    }

public int kthLargestValue(int[][] matrix, int k) {
    int m = matrix.length, n = matrix[0].length;
    int[][] preXor = new int[m + 1][n + 1];
    PriorityQueue<Integer> q = new PriorityQueue<>((a, b) -> a - b);
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            preXor[i][j] = preXor[i - 1][j] ^ preXor[i][j - 1] ^ preXor[i - 1][j - 1] ^ matrix[i - 1][j - 1];
            if (q.size()<k) q.offer(preXor[i][j]);
            else{
                if (q.peek() < preXor[i][j]) {
                    q.poll();
                    q.offer(preXor[i][j]);
                }
            }
        }
    }
    return q.peek();
}

public int maxAbsoluteSum(int[] nums) {
    // 计算前缀和,找出最大值和最小值
    // 结果为max(最大值-最小值,|最大值|,|最小值|)
    int len = nums.length;
    int sum = nums[0], min = nums[0], max = nums[0];
    for (int i = 1; i < len; i++) {
        sum += nums[i];
        min = Math.min(min, sum);
        max = Math.max(max, sum);
    }
    return Math.max((max - min), Math.max(Math.abs(min), Math.abs(max)));
}

    int len = candiesCount.length;
    long[] preSum = new long[len + 1];
    for (int i = 1; i <= len; i++) preSum[i] = preSum[i - 1] + candiesCount[i - 1];

    boolean[] ans = new boolean[queries.length];
    for (int i = 0; i < queries.length; i++) {
        int type = queries[i][0], day = queries[i][1] + 1, limit = queries[i][2];
        long min = preSum[type] + 1;
        long max = preSum[type + 1];
        if ((long) day * limit < min) ans[i] = false;
        else ans[i] = day <= max;
    }
    return ans;
}
1109. 航班预订统计
其实题目的意思是，给你输入一个长度为 n 的数组 nums，索引表示航班号，索引对应的数据表示是该航班要多少个座位。输入一个 bookings，里面是若干三元组 (i,j,k)，每个三元组的含义就是要求你给 nums 数组的闭区间 [i-1,j-1] 中所有元素都加上 k。请你返回最后的 nums 数组是多少？
PS：因为题目说的 n 是从 1 开始计数的，而数组索引从 0 开始，所以对于输入的三元组 (i,j,k)，数组区间应该对应 [i-1,j-1]。
public int[] corpFlightBookings(int[][] bookings, int n) {
    int[] diff = new int[n];
    for (int[] booking : bookings) {
        int lo = booking[0], hi = booking[1], sit = booking[2];
        diff[lo - 1] += sit;
        // 避免数组越界
        if (hi != n) diff[hi] -= sit;
    }

    int[] ans = new int[n];
    ans[0] = diff[0];
    for (int i = 1; i < n; i++) {
        ans[i] = ans[i - 1] + diff[i];
    }
    return ans;
}
