public boolean containsNearbyAlmostDuplicate(int[] nums, int k, int t) {
    // |nums[i] - nums[j]| <= t
    // nums[j]-nums[i]>=-t && nums[j]-nums[i]<=t
    // nums[j]+t>=nums[i] && nums[j]-t<=nums[i];
    int len = nums.length;
    TreeMap<Long, Integer> tree = new TreeMap<>();

    for (int lo = 0, hi = 0; hi < len; hi++) {
        if (hi - lo > k) tree.remove((long) nums[lo++]);
        long num = nums[hi];
        Map.Entry<Long, Integer> entry = tree.floorEntry(num + t);
        if (entry != null && Math.abs(hi - entry.getValue()) <= k && num - t <= entry.getKey()) return true;
        tree.put(num, hi);
    }
    return false;
}

private void absSum(List<Integer> nums, long[] ans) {

        sum[i] = sum[i - 1] + nums.get(i - 1);

        long a = (long) (i + 1) * nums.get(i) - (sum[i + 1] - sum[0]);

public int kthLargestValue(int[][] matrix, int k) {
    int m = matrix.length, n = matrix[0].length;
    int[][] preXor = new int[m + 1][n + 1];
    PriorityQueue<Integer> q = new PriorityQueue<>((a, b) -> a - b);
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            preXor[i][j] = preXor[i - 1][j] ^ preXor[i][j - 1] ^ preXor[i - 1][j - 1] ^ matrix[i - 1][j - 1];
            if (q.size()<k) q.offer(preXor[i][j]);
            else{
                if (q.peek() < preXor[i][j]) {
                    q.poll();
                    q.offer(preXor[i][j]);
                }
            }
        }
    }
    return q.peek();
}
