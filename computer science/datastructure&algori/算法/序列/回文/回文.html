  public String longestPalindrome(String s) {
      int len = s.length();
      if (len < 2) {
          return s;
      }

      int curLen = 0;
      int maxLen = 1;
      int maxLenStart = 0;
      for (int i = 1; i < len; ++i) {
          char c = s.charAt(i);
          int lo = i - 1, hi = i + 1;
          while (lo >= 0 && s.charAt(lo) == c) lo--;
          while (hi < len && s.charAt(hi) == c) hi++;
          while (lo >= 0 && hi < len && s.charAt(lo) == s.charAt(hi)) {
              lo--;
              hi++;
          }
          curLen = hi - lo - 1;
          if (maxLen < curLen) {
              maxLen = curLen;
              maxLenStart = lo + 1;
          }
      }

      return s.substring(maxLenStart, maxLenStart + maxLen);
  }

  public int longestPalindromeSubseq(String s) {
      int len = s.length();
      int[][] dp = new int[len][len];
      for (int i = len - 1; i >= 0; i--) {
          dp[i][i] = 1;
          for (int j = i + 1; j < len; j++) {
              dp[i][j] = s.charAt(i) == s.charAt(j) ? dp[i + 1][j - 1] + 2

          }
      }
      return dp[0][len - 1];
  }
