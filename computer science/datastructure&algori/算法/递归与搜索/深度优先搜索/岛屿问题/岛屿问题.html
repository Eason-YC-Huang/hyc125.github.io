public int numIslands(char[][] grid) {
        int ans = 0;
        Set<String> seen = new HashSet<>();
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1' && !seen.contains(i + "-" + j)) {
                    dfs(grid, i, j, seen);
                    ans++;
                }
            }
        }
        return ans;
    }

    public void dfs(char[][] grid, int i, int j, Set<String> seen) {
        if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length) return;

        String key = i + "-" + j;
        if (grid[i][j] != '1' || seen.contains(key)) return;

        seen.add(key);
        dfs(grid, i + 1, j, seen);
        dfs(grid, i, j + 1, seen);
        dfs(grid, i - 1, j, seen);
        dfs(grid, i, j - 1, seen);
    }

public int numIslands(char[][] grid) {
    int ans = 0;
    for (int i = 0; i < grid.length; i++) {

            if (grid[i][j] == '1') {
                dfs(grid, i, j);
                ans++;
            }
        }
    }

}

public void dfs(char[][] grid, int i, int j) {
    if (i < 0 || j < 0 || i >= grid.length || j >= grid[0].length) return;

    if (grid[i][j] != '1') return;

    grid[i][j] = '2';
    dfs(grid, i + 1, j);
    dfs(grid, i, j + 1);
    dfs(grid, i - 1, j);
    dfs(grid, i, j - 1);
}

public int islandPerimeter(int[][] grid) {
    for(int i=0;i<grid.length;i++)
        for(int j=0;j<grid[0].length;j++)
            if(grid[i][j]==1)
                // 因为只有一个岛屿，处理一个陆地后，即可立即返回结果
                return dfs(grid,i,j);
    return 0;
}

// 一个区域的周长的其四周非陆地区域的个数
public int dfs(int[][] grid, int i , int j){
    if(i<0||j<0||i>=grid.length||j>=grid[0].length) return 1;
    if(grid[i][j]==0) return 1;
    if(grid[i][j]==2) return 0;
    grid[i][j] = 2;
    return dfs(grid,i,j+1)
         + dfs(grid,i,j-1)
         + dfs(grid,i+1,j)
         + dfs(grid,i-1,j);
}
695. 最大的岛屿面积
当前岛屿的面积等于当前区域的面积+周围区域的面积，记得避免重复统计。
public int maxAreaOfIsland(int[][] grid) {
    int ans = 0;
    for(int i=0;i<grid.length;i++)
        for(int j=0;j<grid[0].length;j++)
            if(grid[i][j]==1)
                ans = Math.max(ans,dfs(grid,i,j));
    return ans;
}

public int dfs(int[][] grid, int i, int j){
    if(i<0||j<0||i>=grid.length||j>=grid[0].length) return 0;
    if(grid[i][j] != 1) return 0;

    grid[i][j] = 2;
    return 1
           + dfs(grid,i,j+1)
           + dfs(grid,i,j-1)
           + dfs(grid,i+1,j)
           + dfs(grid,i-1,j);
}
1254. 统计封闭岛屿的数目
岛屿当前区域的其它区域都是封闭的话，则该岛屿是封闭岛屿
public int closedIsland(int[][] grid) {
    int ans = 0;
    for(int i=0;i<grid.length;i++)
        for(int j=0;j<grid[0].length;j++)
            if(grid[i][j]==0&&dfs(grid,i,j))
                ans++;
    return ans;
}

public boolean dfs(int[][] grid, int i, int j){
    if(i<0||j<0||i>=grid.length||j>=grid[0].length) return false;
    if(grid[i][j]!=0) return true;
    grid[i][j] = 2;
    // 注意这里不能用&&
    // &&的短路功能，会导致当前岛屿的剩余区域没有被访问
    return dfs(grid,i,j+1)
         & dfs(grid,i,j-1)
         & dfs(grid,i+1,j)
         & dfs(grid,i-1,j);
}

public int countSubIslands(int[][] grid1, int[][] grid2) {
    int ans = 0;
    for(int i=0;i<grid2.length;i++)
        for(int j=0;j<grid2[0].length;j++)
            if(grid2[i][j]==1&&dfs(grid1,grid2,i,j))
                ans++;
    return ans;
}

// grid2岛屿的当前区域及其周围区域，是否是grid1岛屿的子区域
public boolean dfs(int[][] grid1, int[][] grid2, int i, int j){
    if(i<0||j<0||i>=grid2.length||j>=grid2[0].length) return true;

    if(grid2[i][j]==1&&grid1[i][j]==0) return false;
    if(grid2[i][j]!=1) return true;
    grid2[i][j] = 2;

    return dfs(grid1,grid2,i,j+1)
         & dfs(grid1,grid2,i,j-1)
         & dfs(grid1,grid2,i+1,j)
         & dfs(grid1,grid2,i-1,j);
}
827. 最大人工岛
遍历所有陆地,统计所有岛屿的面积，遍历所有海洋,统计当前海洋与周围不同岛屿的面积。
public int largestIsland(int[][] grid) {
    Map<Integer, Integer> area = new HashMap<>();
    int id = 2;
    int ans = 0;
    for (int i = 0; i < grid.length; i++)
        for (int j = 0; j < grid[0].length; j++)
            if (grid[i][j] == 1) {
                int a = dfs(grid, i, j, id);
                ans = Math.max(ans, a);
                area.put(id, a);
                id++;
            }

    for (int i = 0; i < grid.length; i++)
        for (int j = 0; j < grid[0].length; j++)
            if (grid[i][j] == 0) {
                Set<Integer> neighbor = new HashSet<>();
                if (!outOfArea(grid, i, j + 1) && grid[i][j + 1] > 1) neighbor.add(grid[i][j + 1]);
                if (!outOfArea(grid, i, j - 1) && grid[i][j - 1] > 1) neighbor.add(grid[i][j - 1]);
                if (!outOfArea(grid, i + 1, j) && grid[i + 1][j] > 1) neighbor.add(grid[i + 1][j]);
                if (!outOfArea(grid, i - 1, j) && grid[i - 1][j] > 1) neighbor.add(grid[i - 1][j]);
                ans = Math.max(ans, neighbor.stream().mapToInt(area::get).sum() + 1);
            }
    return ans;
}

// 统计该到岛屿的面积,并把所有区域标记为id
private int dfs(int[][] grid, int i, int j, int id) {
    if (outOfArea(grid, i, j)) return 0;
    if (grid[i][j] != 1) return 0;
    grid[i][j] = id;

    return 1
            + dfs(grid, i, j + 1, id)
            + dfs(grid, i, j - 1, id)
            + dfs(grid, i + 1, j, id)
            + dfs(grid, i - 1, j, id);
}

private boolean outOfArea(int[][] grid, int i, int j) {
    return i < 0 || j < 0 || i >= grid.length || j >= grid[0].length;
}
